CREATE TABLE ARTICLE_CATEGORY(
   CATEGORY_ID INT,
   ARTICLE_CATEGORY_NAME VARCHAR(60),
   PRIMARY KEY(CATEGORY_ID)
);

CREATE TABLE DISCOUNT(
   DISCOUNT_ID INT,
   NAME VARCHAR(50),
   PERCENT INT NOT NULL,
   PRIMARY KEY(DISCOUNT_ID)
);

CREATE TABLE CLIENT_CATEGORY(
   CATEGORY_ID INT,
   TITLE VARCHAR(50) NOT NULL,
   PRIMARY KEY(CATEGORY_ID)
);

CREATE TABLE LOOK(
   LOOK_ID INT,
   LOOK_TITLE VARCHAR(50),
   PRIMARY KEY(LOOK_ID)
);

CREATE TABLE INTEREST(
   INTEREST_ID INT,
   INTEREST_NAME VARCHAR(50),
   PRIMARY KEY(INTEREST_ID)
);

CREATE TABLE SOCIAL_MEDIA(
   SOCIAL_MMEDIA_ID INT,
   SMEDIA_TITLE VARCHAR(50) NOT NULL,
   PRIMARY KEY(SOCIAL_MMEDIA_ID)
);

CREATE TABLE PLACE(
   ID_PLACE INT,
   PLACE_TITLE VARCHAR(100) NOT NULL,
   PLACE_STREET_NUMBER INT NOT NULL,
   PLACE_COUNTRY VARCHAR(50) NOT NULL,
   PLACE_STREET_LABEL VARCHAR(100) NOT NULL,
   PLACE_ZIP_CODE INT NOT NULL,
   PRIMARY KEY(ID_PLACE)
);

CREATE TABLE IMPORTANT_DATE(
   IMPORTANT_CLIENT_DATE_ID INT,
   IMPORTANT_DATE_TITLE DATE NOT NULL,
   IMPORTANT_CLIENT_DATE_COMMMENT TEXT,
   PRIMARY KEY(IMPORTANT_CLIENT_DATE_ID)
);

CREATE TABLE SIZE(
   SIZE_ID INT,
   SIZE_TITLE VARCHAR(100),
   PRIMARY KEY(SIZE_ID)
);

CREATE TABLE CLIENT(
   CLIENT_ID INT,
   FIRST_NAME VARCHAR(100) NOT NULL,
   LAST_NAME VARCHAR(100) NOT NULL,
   SEXE VARCHAR(1) NOT NULL,
   BIRTHDAY DATE,
   OTHER_DATES DATE,
   EMAIL VARCHAR(100),
   TEL VARCHAR(120),
   COMMENTS TEXT,
   CLIENT_SOCIAL_MEDIA_ID INT,
   CLIENT_SIZE_ID INT,
   CATEGORY_ID INT NOT NULL,
   PRIMARY KEY(CLIENT_ID),
   FOREIGN KEY(CATEGORY_ID) REFERENCES CLIENT_CATEGORY(CATEGORY_ID)
);

CREATE TABLE WISH(
   WISH_ID INT,
   WISH_TAG_ID INT,
   EXECUTED_DATE DATE,
   PLACED_DATE DATE NOT NULL,
   COMMENTS TEXT,
   WISHED_SIZE VARCHAR(50),
   WISHED_COLOR VARCHAR(50),
   CATEGORY_ID INT NOT NULL,
   CLIENT_ID INT NOT NULL,
   PRIMARY KEY(WISH_ID),
   FOREIGN KEY(CATEGORY_ID) REFERENCES ARTICLE_CATEGORY(CATEGORY_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID)
);

CREATE TABLE ADDRESS(
   ID_ADDRESS INT,
   STREET_NUMBER INT NOT NULL,
   STREET_LABEL VARCHAR(80) NOT NULL,
   ZIP_CODE INT NOT NULL,
   COUNTRY VARCHAR(80) NOT NULL,
   IS_MAIN LOGICAL NOT NULL,
   CLIENT_ID INT NOT NULL,
   PRIMARY KEY(ID_ADDRESS),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID)
);

CREATE TABLE TRANSACTION(
   TRANSACTION_ID INT,
   TRANSACTION_TITLE VARCHAR(50) NOT NULL,
   TRANSACTION_STATUS VARCHAR(50) NOT NULL,
   TRANSACTION_DATE DATE,
   CLIENT_ID INT NOT NULL,
   PRIMARY KEY(TRANSACTION_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID)
);

CREATE TABLE ARTICLE(
   ARTICLE_ID INT,
   CODE_ARTICLE VARCHAR(100) NOT NULL,
   CATEGORIE VARCHAR(100) NOT NULL,
   COMMENTS TEXT,
   ARTICLE_STATUS VARCHAR(120) NOT NULL,
   PHOTO_STOCK VARCHAR(700),
   PHOTO_COMMERCE VARCHAR(700),
   SIZE VARCHAR(120),
   SELL_PRICE_HT DECIMAL(15,3),
   COLOR VARCHAR(100),
   TRANSACTION_ID INT,
   CATEGORY_ID INT NOT NULL,
   PRIMARY KEY(ARTICLE_ID),
   FOREIGN KEY(TRANSACTION_ID) REFERENCES TRANSACTION(TRANSACTION_ID),
   FOREIGN KEY(CATEGORY_ID) REFERENCES ARTICLE_CATEGORY(CATEGORY_ID)
);

CREATE TABLE CLIENTS_LOOK(
   CLIENT_ID INT,
   LOOK_ID INT,
   COMMENT TEXT,
   PRIMARY KEY(CLIENT_ID, LOOK_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID),
   FOREIGN KEY(LOOK_ID) REFERENCES LOOK(LOOK_ID)
);

CREATE TABLE CLIENTS_INTEREST(
   CLIENT_ID INT,
   INTEREST_ID INT,
   PRIMARY KEY(CLIENT_ID, INTEREST_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID),
   FOREIGN KEY(INTEREST_ID) REFERENCES INTEREST(INTEREST_ID)
);

CREATE TABLE CLIENTS_SOCIAL_MEDIA_PROFILE(
   CLIENT_ID INT,
   SOCIAL_MMEDIA_ID INT,
   SOCIAL_MEDIA_VALUE VARCHAR(150) NOT NULL,
   PRIMARY KEY(CLIENT_ID, SOCIAL_MMEDIA_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID),
   FOREIGN KEY(SOCIAL_MMEDIA_ID) REFERENCES SOCIAL_MEDIA(SOCIAL_MMEDIA_ID)
);

CREATE TABLE CLIENTS_PLACES(
   CLIENT_ID INT,
   ID_PLACE INT,
   PRIMARY KEY(CLIENT_ID, ID_PLACE),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID),
   FOREIGN KEY(ID_PLACE) REFERENCES PLACE(ID_PLACE)
);

CREATE TABLE OTHER_CLIENTS_DATES(
   CLIENT_ID INT,
   IMPORTANT_CLIENT_DATE_ID INT,
   PRIMARY KEY(CLIENT_ID, IMPORTANT_CLIENT_DATE_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID),
   FOREIGN KEY(IMPORTANT_CLIENT_DATE_ID) REFERENCES IMPORTANT_DATE(IMPORTANT_CLIENT_DATE_ID)
);

CREATE TABLE CLIENT_TRY_ARTICLE(
   ARTICLE_ID INT,
   CLIENT_ID INT,
   ARTICLE_TRY_DATE DATE NOT NULL,
   PRIMARY KEY(ARTICLE_ID, CLIENT_ID),
   FOREIGN KEY(ARTICLE_ID) REFERENCES ARTICLE(ARTICLE_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID)
);

CREATE TABLE CLIENT_SIZE(
   CLIENT_ID INT,
   SIZE_ID INT,
   CLIENT_SIZE_VALUE VARCHAR(50) NOT NULL,
   PRIMARY KEY(CLIENT_ID, SIZE_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID),
   FOREIGN KEY(SIZE_ID) REFERENCES SIZE(SIZE_ID)
);

CREATE TABLE APPLIED_DISCOUNT(
   DISCOUNT_ID INT,
   TRANSACTION_ID INT,
   PRIMARY KEY(DISCOUNT_ID, TRANSACTION_ID),
   FOREIGN KEY(DISCOUNT_ID) REFERENCES DISCOUNT(DISCOUNT_ID),
   FOREIGN KEY(TRANSACTION_ID) REFERENCES TRANSACTION(TRANSACTION_ID)
);
